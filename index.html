    <script>

        // JavaScript for language switching

        function setLanguage(lang) {

            localStorage.setItem('preferredLang', lang); // Save preferred language

            document.querySelectorAll('.lang-content').forEach(content => {

                if (content.getAttribute('data-lang') === lang) {

                    content.style.display = 'block';

                } else {

                    content.style.display = 'none';

                }

            });

            // Update active button state

            document.querySelectorAll('.lang-selector button').forEach(button => {

                if (button.onclick.toString().includes(`setLanguage('${lang}')`)) {

                    button.classList.add('active');

                } else {

                    button.classList.remove('active');

                }

            });

        }


        // On page load, set language from localStorage or default to Vietnamese

        document.addEventListener('DOMContentLoaded', () => {

            const savedLang = localStorage.getItem('preferredLang') || 'vi';

            setLanguage(savedLang);

        });

    </script>
